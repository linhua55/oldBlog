{
  const int i = threadIdx.x;
  dest[i] = a[i] * b[i];
}
""")

    $('aside').each(function(i){
        curOffset=$(this).offset()['top'];
        shiftOffset=0;

            ++shiftOffset;

            elements[i].animate(

                top: offsets[i]
